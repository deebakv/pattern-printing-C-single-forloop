// 1*****1

// 7

#include<stdio.h>
#include<stdlib.h>
#include<string.h>


int main()
{

	int n;
	scanf("%d",&n);
	char _[] = "***********************************************************************************************************";
	int t = strlen(_),z=n-2,j=0;
	for(int i=1;i<n*2;i++)
	{
		if(i<n)
		{
			printf("%s%d%s%d%s\n",_+(t-i+1),i,_+(t-(n*2-i*2)+1),i,_+(t-i+1));
		}
		else if(i==n)
		{
			printf("%s%d%s\n",_+(t-i+1),i,_+(t-i+1));
		}
		else
		{
			printf("%s%d%s%d%s\n",_+(t-z),n-j-1,_+(t-(j*2)-1),n-j-1,_+(t-z));
			--z;++j;
		}
	}
}
